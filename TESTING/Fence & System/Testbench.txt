`timescale 1ns / 1ps

module cpu_tb;
    reg clk, reset;
    cpu dut(.clk(clk), .reset(reset));

    integer errors;

    task check_reg;
        input integer idx;
        input [31:0] expected;
        begin
            if (dut.rf.rf[idx] !== expected) begin
                $display("REG FAIL x%0d got=%h expected=%h", idx, dut.rf.rf[idx], expected);
                errors = errors + 1;
            end
        end
    endtask

    task check_mem;
        input integer idx;
        input [31:0] expected;
        begin
            if (dut.mem.data[idx] !== expected) begin
                $display("MEM FAIL data[%0d] got=%h expected=%h", idx, dut.mem.data[idx], expected);
                errors = errors + 1;
            end
        end
    endtask

    initial begin
        clk = 1;
        reset = 1;
        errors = 0;
        forever #200 clk = ~clk;
    end

    initial begin
        #5 reset = 0;
    end

    initial begin
        repeat (30) @(posedge clk);

        check_reg(1, 32'h00000001);
        check_reg(2, 32'h00000002);
        check_mem(32, 32'h00000002);

        if (errors == 0) $display("PASS: Fence/System");
        else $display("FAIL: Fence/System, errors=%0d", errors);

        $finish;
    end
endmodule
